Advanced Aggregation Functions

Part 1: sum() - Total Calculations

Essential Analytics Functions

Master these aggregation functions for customer analytics

// Calculate total commission earned
MATCH (a:Agent)-[:EARNED_COMMISSION]->(c:Commission)
RETURN a.agent_id,
       sum(c.commission_amount) AS total_earned

// Total premium by risk tier
MATCH (c:Customer)-[:HOLDS_POLICY]->(p:Policy)
RETURN c.risk_tier,
       sum(p.annual_premium) AS total_premium
ORDER BY total_premium DESC

Advanced Aggregation Functions

Part 2: avg() - Average Calculations

// Average commission per agent
MATCH (a:Agent)-[:EARNED_COMMISSION]->(c:Commission)
RETURN a.agent_id,
       avg(c.commission_amount) AS avg_commission,
       count(c) AS commission_count

// Average policy premium by segment
MATCH (c:Customer)-[:HOLDS_POLICY]->(p:Policy)
RETURN c.customer_segment,
       avg(p.annual_premium) AS avg_premium

Commission Analytics Patterns

Part 1: Commission Aggregation

// Complete commission analysis
MATCH (a:Agent)-[:EARNED_COMMISSION]->(c:Commission)
RETURN a.agent_id AS agent,
       count(c) AS total_commissions,
       sum(c.commission_amount) AS total_earned,
       avg(c.commission_amount) AS avg_commission,
       min(c.commission_amount) AS min_commission,
       max(c.commission_amount) AS max_commission
ORDER BY total_earned DESC

Commission Analytics Patterns

Part 2: round() for Clean Output

// Round financial values
MATCH (a:Agent)-[:EARNED_COMMISSION]->(c:Commission)
WITH a,
     sum(c.commission_amount) AS total,
     avg(c.commission_amount) AS average
RETURN a.agent_id,
       round(total * 100) / 100 AS total_rounded,
       round(average * 100) / 100 AS avg_rounded

Graph Path Functions - shortestPath()

Part 1: Basic shortestPath Syntax

Finding Shortest Paths in Neo4j

The shortestPath() function finds the shortest path between two nodes

// Find shortest path between two customers
MATCH (c1:Customer {customer_number: 'C001'}),
      (c2:Customer {customer_number: 'C050'})
MATCH path = shortestPath(
  (c1)-[*..5]-(c2)
)
RETURN path, length(path) AS hops

// Shortest path with relationship type
MATCH (c1:Customer), (c2:Customer)
WHERE c1.customer_number = 'C001'
  AND c2.customer_number = 'C100'
MATCH path = shortestPath(
  (c1)-[:REFERS*]-(c2)
)
RETURN path

Graph Path Functions - shortestPath()

Part 2: Path Length and Analysis

// Analyze connection distances
MATCH (c1:Customer), (c2:Customer)
WHERE c1 <> c2
MATCH path = shortestPath((c1)-[*..5]-(c2))
WITH length(path) AS distance,
     count(*) AS pair_count
RETURN distance, pair_count
ORDER BY distance

// Find all paths of specific length
MATCH (c1:Customer)-[*3]-(c2:Customer)
WHERE c1 <> c2
RETURN count(DISTINCT [c1, c2]) AS paths

Variable-Length Path Patterns

Part 1: Variable-Length Syntax

// Exactly 3 hops
MATCH (c1)-[*3]-(c2)

// 1 to 3 hops (min..max)
MATCH (c1)-[*1..3]-(c2)

// Up to 5 hops
MATCH (c1)-[*..5]-(c2)

// At least 2 hops
MATCH (c1)-[*2..]-(c2)

Variable-Length Path Patterns

Part 2: Network Analysis Patterns

// Find referral networks
MATCH (c:Customer)-[:REFERS*1..3]->(referred:Customer)
RETURN c.customer_number AS referrer,
       count(DISTINCT referred) AS network_size

// Degree centrality calculation
MATCH (c:Customer)-[r]-()
WITH c, count(r) AS degree
RETURN avg(degree) AS avg_centrality,
       max(degree) AS max_centrality

Temporal Data Management

Essential functions for compliance reporting and audit trails

// Current date
RETURN date() AS today

RETURN date('2024-01-15') AS specific_date

// Date from components
RETURN date({year: 2024, month: 1, day: 15}) AS built_date

Date and Duration Functions

Part 2: duration() Function

// Create durations
RETURN duration({days: 365}) AS one_year
RETURN duration({months: 6}) AS six_months
RETURN duration({years: 1, months: 6}) AS complex

// Duration between dates
MATCH (p:Policy)
RETURN p.policy_number,
       duration.between(
         p.effective_date,
         date()
       ).days AS days_active

Compliance Temporal Queries

Part 1: Time-Based Filtering

// Policies effective in last year
MATCH (p:Policy)
WHERE p.effective_date >= date() - duration({days: 365})
RETURN count(p) AS recent_policies

// Audit trail time windows
MATCH (n)
WHERE n.created_at >= date() - duration({months: 3})
  AND n.created_at <= date()
RETURN count(n) AS recent_changes

Compliance Temporal Queries

Part 2: Version Tracking with Dates

// Track entity versions by date
MATCH (n)
WHERE n.version IS NOT NULL
  AND n.updated_at IS NOT NULL
WITH n.entity_type AS type,
     n.version AS version,
     n.updated_at AS updated
RETURN type, version, updated
ORDER BY updated DESC

// Audit compliance reporting
MATCH (n)
WHERE n.created_at IS NOT NULL
RETURN n.created_by AS source,
       date(n.created_at) AS date,
       count(n) AS entities_created

Machine Learning Feature Engineering

Part 1: Feature Extraction Patterns

Preparing Data for Predictive Models

Feature engineering transforms raw graph data into ML-ready features

// Extract customer features for ML
MATCH (c:Customer)
OPTIONAL MATCH (c)-[:HOLDS_POLICY]->(p:Policy)
OPTIONAL MATCH (c)-[:FILED_CLAIM]->(claim:Claim)
WITH c,
     count(DISTINCT p) AS policy_count,
     sum(p.annual_premium) AS total_premium,
     count(DISTINCT claim) AS claim_count,
     sum(claim.claim_amount) AS total_claims,
     duration.between(c.join_date, date()).years AS tenure_years
RETURN c.customer_number,
       policy_count,
       total_premium,
       claim_count,
       total_claims,
       tenure_years,
       c.credit_score

Machine Learning Feature Engineering

Part 2: Behavioral Features

// Network-based features
MATCH (c:Customer)
OPTIONAL MATCH (c)-[:REFERS]->(referred:Customer)
OPTIONAL MATCH (c)<-[:REFERS]-(referrer:Customer)
OPTIONAL MATCH (c)-[r]-()
WITH c,
     count(DISTINCT referred) AS referral_count,
     count(DISTINCT referrer) AS referred_by_count,
     count(r) AS total_connections
RETURN c.customer_number,
       referral_count,
       referred_by_count,
       total_connections AS network_degree

Model Confidence and Prediction Scoring

Part 1: Confidence Metrics

// Model confidence scoring
MATCH (c:Customer)-[:HAS_PREDICTION]->(pm:PredictiveModel)
RETURN c.customer_number,
       pm.churn_probability,
       pm.model_confidence,
       CASE
         WHEN pm.model_confidence > 0.8 THEN 'High Confidence'
         WHEN pm.model_confidence > 0.6 THEN 'Medium Confidence'
         ELSE 'Low Confidence'
       END AS confidence_tier

// Filter by confidence threshold
MATCH (c:Customer)-[:HAS_PREDICTION]->(pm:PredictiveModel)
WHERE pm.model_confidence > 0.7
  AND pm.churn_probability > 0.5
RETURN c.customer_number,
       pm.churn_probability,
       pm.retention_actions

Model Confidence and Prediction Scoring

Part 2: Prediction Quality Assessment

// Assess prediction quality
MATCH (pm:PredictiveModel)
WITH avg(pm.model_confidence) AS avg_confidence,
     count(pm) AS total_predictions,
     sum(CASE WHEN pm.model_confidence > 0.7
         THEN 1 ELSE 0 END) AS high_confidence_count
RETURN avg_confidence,
       total_predictions,
       high_confidence_count,
       round(high_confidence_count * 100.0
             / total_predictions) AS high_confidence_pct

// Model performance by segment
MATCH (c:Customer)-[:HAS_PREDICTION]->(pm:PredictiveModel)
RETURN c.customer_segment,
       avg(pm.model_confidence) AS avg_confidence,
       count(pm) AS prediction_count
ORDER BY avg_confidence DESC
